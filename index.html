from js import document from io import BytesIO import pandas as pd from pyscript import when @when('change', '#upload') async def processFile(*args): csv_file = document.getElementById('upload').files.item(0) array_buf = await csv_file.arrayBuffer() # Get arrayBuffer from file file_bytes = array_buf.to_bytes() # convert to raw bytes array csv_file = BytesIO(file_bytes) # Wrap in Python BytesIO file-like object # Read the CSV file into a Pandas DataFrame df = pd.read_csv(csv_file) dataArray = df.to_numpy() print(dataArray) # Convert each row to a list and store them in an array rows_array = df.values.tolist() # Now, 'rows_array' contains each row of the CSV file as a list print(rows_array)
